NBNBNB: Наследующий класс можно кастить к любому наследуемому классу.
С методами в наследовании работают точно так же как и с полями:

struct base {
	void f();
	void g();
};
struct derived : base {
	void g();
	void h();
};
int main() {
	base b;
	b.f();
	b.g();
	derived d;
	d.f();
	d.g(); // запустит g() из derived
	d.h();
	d.base::g(); // запустит g() из base
}
